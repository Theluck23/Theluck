/*判断任意三个大于0的实数能否构成三角形，并判断是否为等边，等腰，直角。还需求出三角形面积
制作人：谢建明
*/ 
#include<stdio.h>
#include<math.h>//后面会用到数学公式 
int main()
{
	double a,b,c,p;//定义变量 
	printf("输入三边边长,空格分开:\n");
	scanf("%lf%lf%lf",&a,&b,&c);//输入三边边长 
	p=(a+b+c)/2;//求出周长的一半，后面求面积海伦公式会用到 
	if((a+b)>c&&(b+c)>a&&(a+c)>b)//“三角形的任意两条边之和大于第三边”用于判断是否能构成三角形 
	{//如果能构成三角形，则进行以下步骤 
		printf("能构成三角形\n"); 
		printf("该三角形的面积是%lf\n",sqrt(p*(p-a)*(p-b)*(p-c)));//利用海伦公式求出面积，并打印出（利用了math.h中的sqrt（）函数） 
		if(a==b&&b==c)//判断是否为等边三角形 
		{
			printf("该三角形为等边三角形\n");
		}
		else if(a==b||b==c||c==a)//在不是等边三角形的情况下，判断是否为等腰三角形 
		{
			printf("该三角形为等腰三角形\n");
		}
		if((pow(a,2)+pow(b,2)==pow(c,2))||(pow(a,2)==pow(b,2)+pow(c,2))||(pow(a,2)+pow(c,2)==pow(b,2)))//利用勾股定理判断是否为直角三角形 
		{                                                                                              //用到了math.h中的pow（）函数 
			printf("该三角形为直角三角形\n");
		}
	}
	else printf("不能构成三角形");//不能构成三角形 
 }
